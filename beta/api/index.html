StateManager / State


Для работы стора используем внешний метод state<type>(defaulValue).
state являеться объектом билдером, в нём содержаться разные функции, для работы с пропами, запросами на сервер, и постоянной памяти Стора.

Пример декларации: 
import StateManager, { state } from "nuxoblivius"

export default class MyStore extends StateManager {
    // =======================================================
    // Здесь мы создаём объект стора
    // В данном примере обычное текстовое реактивное поле
    // И помечаем что поле не массив
    public testStore = state<string>('test')
        .one()
    // =======================================================

    constructor() {
        super("My Store")
        this.manage() // Запуск создание стора
    }
}

StateManager / Basic Prop


Для создания простого реактивного поля в state.

Мы используем цепочку state().one() или state().many().
Методы one и many отвечают за точного ответа являеться ли объект массивом. Во всех структурах one и many завершают билдер и возвращают ответ в виде готового объекта.

Пример декларации: 
import StateManager, { state } from "nuxoblivius"

export default class Player extends StateManager {
    // =======================================================
    // Здесь мы создаём объект стора
    // В данном примере что у игрока в руке
    // И помечаем что поле не массив
    public currentItem = state<string>('none')
        .one()
    // Здесь мы создаём объект стора
    // В данном примере инвентарь игрока
    // И помечаем что поле массив
    public inventory = state<string>(['stick'])
        .many()
    // =======================================================

    constructor() {
        super("Player")
        this.manage() // Запуск создание стора
    }
}
